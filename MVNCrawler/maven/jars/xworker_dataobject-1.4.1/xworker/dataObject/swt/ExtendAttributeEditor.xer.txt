import org.xmeta.Thing;

import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.TreeItem;

def treeItem = thingContext.get("outlineTree").getSelection()[0];
for(item in dataTable.getItems()){
    def data = item.getData();
    if(item.getChecked()){
        //已存在修改，不存在增加
        def have = false;
        for(attr in thing.getChilds()){
            if(attr.metadata.name == data.name && attr.getThingName() == data.type){
                have = true;                
            }
        }
        if(!have){            
            def extThing = world.getThing(data.path);
            if(extThing != null){
                def attr = extThing.detach();
                thing.addChild(attr);
                
                //更新Outline
                def childTreeItem = new TreeItem(treeItem, SWT.NONE);
                childTreeItem.setText(attr.metadata.label + " (" + attr.thingName + ")");
                childTreeItem.setData(attr);
            }
        }
    }else{
        //删除对应的属性
        for(attr in thing.getChilds()){
            if(attr.metadata.name == data.name && attr.getThingName() == data.type){
                thing.removeChild(attr);
                for(childItem in treeItem.getItems()){
                    if(childItem.getData() == attr){
                        childItem.dispose();
                    }
                }
                break;
            }
        }
    }
}

if(thingContext.get("okButtonSelection") != null){
    thingContext.okButtonSelection.handleEvent(null);
}else{
    thing.save();
}
shell.dispose();

import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.MessageBox;

def extendDataObject = world.getThing(thing.extendedDataObject);
if(extendDataObject == null){
    def box = new MessageBox(shell, SWT.ICON_INFORMATION | SWT.OK);
    box.setText("提示");
    box.setMessage("没有继承数据对象。");
    box.open();
    shell.setData("menuShellDispose", true);
    return;
}

def mappingFields = [];
for(child in extendDataObject.getChilds()){
    def data = [:];
    data.name = child.metadata.name;
    data.label = child.metadata.label;
    data.type = child.getThingName();
    data.checked = false;
    data.thing = child;
    data.path = child.metadata.path;
    mappingFields.add(data);
}
for(field in mappingFields){
    for(child in thing.getChilds()){
        if(field.name == child.metadata.name && field.type == child.getThingName()){
            field.checked = true;
        }
    }
}

//初始化表格
actionContext.getScope(0).put("attributes", mappingFields);
dataStore.doAction("load", actionContext);
