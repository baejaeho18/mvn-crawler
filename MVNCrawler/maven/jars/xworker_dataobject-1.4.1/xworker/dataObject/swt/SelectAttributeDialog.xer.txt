import org.xmeta.util.UtilString;

import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.MessageBox;

//参数列表
def params = UtilString.getParams(param, ",");
def indicator = params.dataObjectIndicator;
def attributeThing = null;
if(indicator == null || indicator == "" || indicator == "self"){
    attributeThing = parentContext.thing.parent;
}else if(indicator.startsWith("action:")){
    def action = indicator.substring(7, indicator.length());
    attributeThing = parentContext.thing.doAction(action, parentContext);
}else{
    def values = parentContext.model.doAction("getValue", parentContext);
    //log.info("values=" + values);
    attributeThing = world.getThing(values.get(indicator));
}
if(attributeThing == null){
    return;
}

def attributes = [];
for(attribute in attributeThing.get("attribute@")){
    def attr = [:];
    attr.name = attribute.metadata.name;
    attr.label = attribute.metadata.label;
    attributes.add(attr);
}

def context = [:];
for(child in attributeThing.get("thing@")){
    def childThing = world.getThing(child.dataObjectPath);
    if(childThing == null){
        continue;
    }
    
    addThingAttribute(child.metadata.name, childThing, attributes, context);
}
actionContext.getScope(0).put("attributes", attributes);
dataStore.doAction("load", actionContext);

def addThingAttribute(name, thing, attributes, context){
    if(context.get(thing.metadata.path) != null){
        return;
    }
    
    context.put(thing.metadata.path, thing);
    
    for(attribute in thing.get("attribute@")){
        def attr = [:];
        attr.name = name + "." + attribute.metadata.name;
        attr.label = attribute.metadata.label;
        attributes.add(attr);
    }
    
    for(child in thing.get("thing@")){
        def childThing = world.getThing(child.dataObjectPath);
        if(childThing == null){
            continue;
        }
        addThingAttribute(name + "." + childThing.metadata.name, childThing, attributes, context);
    }
}
